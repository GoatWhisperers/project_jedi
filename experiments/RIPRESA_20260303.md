Script completo da usare:
```bash
cat > /tmp/run_gemma2_ripresa.sh << 'SCRIPT'
#!/bin/bash
PYTHON="/home/lele/codex-openai/project_jedi/.venv/bin/python"
SCRIPTS="/home/lele/codex-openai/project_jedi/scripts"
ROOT="/home/lele/codex-openai/project_jedi"
LOG_DIR="/tmp/decompose_gemma2_ripresa_$(date +%Y%m%d_%H%M%S)"
MAIN_LOG="$LOG_DIR/batch_main.log"
mkdir -p "$LOG_DIR"
log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" | tee -a "$MAIN_LOG"; }

CONCEPTS=("secco_vs_umido:sensoriale" "duro_vs_morbido:sensoriale" "rumore_vs_silenzio:uditivo" "dolce_vs_amaro:gustativo" "odore_forte_vs_inodore:olfattivo")

log "Caricamento Gemma2-Uncensored..."
curl -sf -X POST "http://localhost:8010/api/load_model" -H "Content-Type: application/json" -d '{"name":"Gemma2-Uncensored"}' > /dev/null
sleep 90

for CC in "${CONCEPTS[@]}"; do
    C="${CC%%:*}"; CAT="${CC##*:}"
    log "── Gemma2-Uncensored / $C ──"
    "$PYTHON" "$SCRIPTS/decompose.py" --concept "$C" --model "Gemma2-Uncensored" \
        --category "$CAT" --max-depth 0 --max-iter 2 \
        --steering-url http://localhost:8010 --m40-url http://localhost:11435 \
        > "$LOG_DIR/${C}.log" 2>&1
    [ $? -eq 0 ] && log "✓ OK $(grep VERDETTO $LOG_DIR/${C}.log | tail -1)" || log "✗ FAILED"
done

"$PYTHON" "$SCRIPTS/build_catalog_multi.py" >> "$MAIN_LOG" 2>&1
log "Catalog aggiornato."

cd "$ROOT"
git add output/vector_library output/decompose_runs output/sub_concept_evals output/catalog.json config/sub_concepts 2>/dev/null || true
git diff --cached --quiet || git commit -m "data: decompose Gd1 Gemma2 batch completo (9/9)

Co-Authored-By: Claude Sonnet 4.6 <noreply@anthropic.com>"
GIT_SSH_COMMAND="ssh -i /home/lele/.ssh/id_ed25519_github" git push origin main
SCRIPT
chmod +x /tmp/run_gemma2_ripresa.sh
nohup bash /tmp/run_gemma2_ripresa.sh > /tmp/gemma2_ripresa.log 2>&1 &
echo "Batch PID: $!"
```
